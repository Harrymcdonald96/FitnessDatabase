Fitness App Database Project - SQL

Showcasing SQL operations for a fitness app database.

STEP 1: Setup Database


CREATE DATABASE FitnessAppDB;
USE FitnessAppDB;

Database 'FitnessAppDB' created and selected for use.

STEP 2: Create Tables


-- Users Table
CREATE TABLE Users (
    user_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    membership_level VARCHAR(50) NOT NULL
);

-- Workouts Table
CREATE TABLE Workouts (
    workout_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    date DATE NOT NULL,
    notes TEXT,
    FOREIGN KEY (user_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

-- Exercises Table
CREATE TABLE Exercises (
    exercise_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    type VARCHAR(50) NOT NULL,
    details TEXT
);

-- Workout_Exercises Table
CREATE TABLE Workout_Exercises (
    workout_exercise_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    workout_id INT NOT NULL,
    exercise_id INT NOT NULL,
    sets INT,
    reps INT,
    weight DECIMAL(5, 2),
    duration TIME,
    distance DECIMAL(5, 2),
    calories_burned INT,
    notes TEXT,
    FOREIGN KEY (workout_id) REFERENCES Workouts(workout_id) ON DELETE CASCADE,
    FOREIGN KEY (exercise_id) REFERENCES Exercises(exercise_id) ON DELETE CASCADE
);

-- Nutrition Table
CREATE TABLE Nutrition (
    nutrition_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    date DATE NOT NULL,
    meal_type VARCHAR(50) NOT NULL,
    food_items TEXT NOT NULL,
    calories INT,
    macronutrients TEXT,
    FOREIGN KEY (user_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

-- Goals Table
CREATE TABLE Goals (
    goal_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    goal_description TEXT NOT NULL,
    target_value DECIMAL(10, 2),
    current_value DECIMAL(10, 2),
    start_date DATE,
    target_date DATE,
    FOREIGN KEY (user_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

-- Achievements Table
CREATE TABLE Achievements (
    achievement_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    achievement_description TEXT NOT NULL,
    date_achieved DATE NOT NULL,
    FOREIGN KEY (user_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

STEP 6: Create View


-- Create a View to Show Users' Workouts and Exercises
CREATE VIEW UserWorkouts AS 
SELECT 
    u.name AS user_name, 
    w.date AS workout_date, 
    e.name AS exercise_name, 
    we.sets, 
    we.reps, 
    we.weight, 
    we.duration, 
    we.distance, 
    we.calories_burned 
FROM Workout_Exercises we 
JOIN Workouts w ON we.workout_id = w.workout_id 
JOIN Users u ON w.user_id = u.user_id 
JOIN Exercises e ON we.exercise_id = e.exercise_id;

STEP 7: Stored Procedures


DELIMITER //

CREATE PROCEDURE LogWorkout (
    IN p_user_id INT,
    IN p_date DATE,
    IN p_notes TEXT
)
BEGIN
    INSERT INTO Workouts (user_id, date, notes)
    VALUES (p_user_id, p_date, p_notes);
END //

DELIMITER ;
                        
CALL LogWorkout(4, '2023-07-11', 'Evening workout');

This would log a workout for the user with ID 4 on '2023-07-11' with the note 'Evening workout'.

STEP 8: CASE Statement


-- Categorize Users Based on Their Membership Level
SELECT 
    user_id, 
    name, 
    email, 
    membership_level, 
    CASE
        WHEN membership_level = 'Free Member' THEN 'Basic Access'
        WHEN membership_level = 'Premium Member' THEN 'Full Access'
        ELSE 'Unknown Level'
    END AS access_level 
FROM Users;
